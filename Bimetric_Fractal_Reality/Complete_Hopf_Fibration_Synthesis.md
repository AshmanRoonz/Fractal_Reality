def fibonacci_decomposition(signal, fundamental_freq, max_n=10):
    """
    Decompose signal into Fibonacci harmonic modes
    
    Args:
        signal: Time series data
        fundamental_freq: Base frequency f_0
        max_n: Maximum Fibonacci index
    
    Returns:
        amplitudes: Complex amplitude for each Fibonacci mode
        frequencies: Corresponding frequencies
    """
    from scipy.fft import fft, fftfreq
    
    # FFT of signal
    N = len(signal)
    yf = fft(signal)
    xf = fftfreq(N, 1/fundamental_freq)

    fibs = fibonacci_sequence(max_n)
    
    # Extract amplitudes at Fibonacci frequencies
    amplitudes = []

        frequencies.append(f_target)
        
        # Find closest FFT bin
        idx = np.argmin(np.abs(xf - f_target))
        amplitudes.append(yf[idx] / N)
    
    return np.array(amplitudes), np.array(frequencies)
