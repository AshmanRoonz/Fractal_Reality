import numpy as np

def higuchi_dimension(data, k_max=10):
    """
    Compute Higuchi fractal dimension
    Reference: Higuchi (1988) Physica D
    """
    N = len(data)
    L = []
    
    for k in range(1, k_max+1):
        Lk = []
        for m in range(k):
            # Length of curve for initial point m
            Lmk = 0
            n_max = int((N-m-1)/k)
            for i in range(1, n_max):
                Lmk += abs(data[m + i*k] - data[m + (i-1)*k])
            Lmk = Lmk * (N-1) / (n_max * k)
            Lk.append(Lmk)
        L.append(np.mean(Lk))
    
    # Fit log(L) vs log(1/k)
    x = np.log(1/np.arange(1, k_max+1))
    y = np.log(L)
    slope, _ = np.polyfit(x, y, 1)
    
    return slope  # This is the fractal dimension

def analyze_gw_event(event_name, detector='H1', 
                      t_start=-0.2, t_dur=0.4):

def box_counting_metric_aware(path, metric_tensor, 
                                epsilon_proper=0.1):
    """
    Box-counting with proper distances from metric
    
    Args:
        path: Array of (x,y,z) coordinates
        metric_tensor: 3x3 spatial metric at each point
        epsilon_proper: Box size in proper distance
    
    Returns:
        fractal_dimension: D from metric-aware measurement
    """
    n_points = len(path)
    scales = np.logspace(-2, 0, 15)  # 15 scales
    counts = []
    
    for epsilon in scales:
        # Occupied boxes set
        boxes = set()
        
        for i, point in enumerate(path):
            # Local metric
            g = metric_tensor[i]
            
            # Coordinate box size from proper distance
            # ε_proper² = g_ij Δx^i Δx^j
            # For diagonal: Δx = ε_proper / √g_ii
            g_spatial = np.diag(g)
            dx = epsilon_proper / np.sqrt(g_spatial)
            
            # Box index (coordinate-dependent)
            box_idx = tuple(np.floor(point / dx).astype(int))
            boxes.add(box_idx)
        
        counts.append(len(boxes))

    x = np.log(1/scales)
    y = np.log(counts)

    return D
